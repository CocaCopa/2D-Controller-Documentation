CharacterCombat.AttackComboCounter

Declaration

public int AttackComboCounter { get; }


Description

Specifies the current combo attack's sequence number.

Attacks can be set up to be part of a combo sequence, where a counter is utilized to designate the attack's sequence number within the combo. This counter is instrumental in tracking the presently executing attack within the combo sequence. Primarily, this property is employed in combination with the PerformNormalAttack function.


Here is how you can access the property. For additional information on invoking the PerformNormalAttack function, refer to the method's documentation. It contains detailed instructions on effectively utilizing this function. 

using System.Collections.Generic;
using UnityEngine;

[RequireComponent(typeof(CharacterCombat))]
public class Example : MonoBehaviour
{
    [SerializeField] private List<AttackSO> comboAttacks;
    
    private CharacterCombat characterCombat;

    private void Awake()
    {
        characterCombat = GetComponent<CharacterCombat>();
    }

    private void Update()
    {
        if (Input.GetKeyDown(KeyCode.C))
        {
            if (characterCombat.AttackComboCounter < comboAttacks.Count)
            {
                characterCombat.PerformNormalAttack(comboAttacks[characterCombat.AttackComboCounter], isPartOfCombo: true);
            }

            Debug.Log("Attack number: " + characterCombat.AttackComboCounter);
        }
    }
}

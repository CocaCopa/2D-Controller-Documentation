CharacterCombat.ChargeTimer

Declaration

public float ChargeTimer { get; }


Description

Indicates the amount of time an attack needs in order to be fully charged in seconds.

If the attack is set up as a chargeable attack, this property monitors the timer according to the charge time specified in the attack's AttackSO scriptable object. This information is valuable for coding visuals that correspond to the attack's charge time.


Here is an example of how you can access the property. 

using UnityEngine;
using CharacterController2D.CombatSystem;

[RequireComponent(typeof(CharacterCombat))]
public class Example : MonoBehaviour
{
    [SerializeField] private AttackSO exampleAttack;
    
    private CharacterCombat characterCombat;

    private void Awake()
    {
        characterCombat = GetComponent<CharacterCombat>();
    }

    private void Update()
    {
        if (Input.GetKey(KeyCode.X))
        {
            characterCombat.PerformChargeAttack(exampleAttack);
        }

        if (characterCombat.IsAttacking && characterCombat.ChargeTimer == 0)
        {
            Debug.Log("The attack is fully charged.");
        }
    }
}

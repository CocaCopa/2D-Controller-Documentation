CharacterCombat.CurrentAttackData

Declaration

public AttackSO CurrentAttackData { get; }


Description

The attack data (scriptable object) of the attack that is currently being performed.

Upon initiating an attack, the relevant data is stored in the CurrentAttackData variable. Accessing this property allows you to retrieve information about the ongoing attack, enabling you to read various details pertaining to the current attack.


Here is how you can access the property. 

using UnityEngine;
using CharacterController2D.CombatSystem;

[RequireComponent(typeof(CharacterCombat))]
public class Example : MonoBehaviour
{
    [SerializeField] private AttackSO exampleAttack;
    
    private CharacterCombat characterCombat;

    private void Awake()
    {
        characterCombat = GetComponent<CharacterCombat>();
    }

    private void Update()
    {
        if (Input.GetKeyDown(KeyCode.C))
        {
            characterCombat.PerformNormalAttack(exampleAttack);
        }

        if (characterCombat.CurrentAttackData != null)
        {
            Debug.Log(characterCombat.CurrentAttackData.AttackName);
        }
    }
}

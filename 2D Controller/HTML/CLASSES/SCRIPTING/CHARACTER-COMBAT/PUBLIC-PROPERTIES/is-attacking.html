CharacterCombat.IsAttacking

Declaration

public bool IsAttacking { get; }


Description

Indicates whether the character is currently attacking.

Upon successful execution of the PerformNormalAttack or PerformChargeAttack functions, the IsAttacking property will return true.


Here is an example of how you can access the property. 

using UnityEngine;
using CharacterController2D.CombatSystem;

[RequireComponent(typeof(CharacterCombat))]
public class Example : MonoBehaviour
{
    [SerializeField] private AttackSO exampleAttack;
    
    private CharacterCombat characterCombat;

    private void Awake()
    {
        characterCombat = GetComponent<CharacterCombat>();
    }

    private void Update()
    {
        if (Input.GetKeyDown(KeyCode.C))
        {
            characterCombat.PerformNormalAttack(exampleAttack);
        }

        Debug.Log("Your character is attacking: " + characterCombat.IsAttacking);
    }
}

HumanoidController.Respawn

Declaration

public virtual void Respawn();


Description

Resets necessary values when the character respawns.


This method needs to be overridden when the character respawns. 

using UnityEngine;

public class Example : HumanoidController
{
    public override void Respawn()
    {
        base.Respawn();

        // vv Your respawn code vv
    }
}

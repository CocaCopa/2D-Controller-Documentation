CharacterDash.DashSpeed

Declaration

public float DashSpeed { get; set; }


Description

The speed of your character while dashing.

This property represents the speed at which your character performs a dash. Adjust it using the component's inspector to set a default value through the 'dashSpeed' field, or modify it dynamically within a script during runtime.


This is an example of how you can use the DashSpeed property in your script: 

using UnityEngine;

[RequireComponent(typeof(CharacterDash))]
public class Example : MonoBehaviour 
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private CharacterEnvironmentalQuery envQuery;
    [Tooltip("Assigned values greater than 1 will result in higher speeds, while values below 1 in lower speeds.")]
    [SerializeField] private float dashSpeedMultiplier;

    private CharacterDash characterDash;

    private float defaultDashSpeed;
    private float characterHeight;
    private bool isGrounded;

    private void Awake()
    {
        characterDash = GetComponent<CharacterDash>();
        
        defaultDashSpeed = characterDash.DashSpeed;
        characterHeight = m_Collider.bounds.size.y;
    }

    private void Update()
    {
        isGrounded = envQuery.GroundCheck();
        characterDash.DashSpeed = defaultDashSpeed * dashSpeedMultiplier;

        if (Input.GetKeyDown(KeyCode.LeftShift))
        {
            characterDash.Dash(characterHeight, isGrounded);
        }
    }
}

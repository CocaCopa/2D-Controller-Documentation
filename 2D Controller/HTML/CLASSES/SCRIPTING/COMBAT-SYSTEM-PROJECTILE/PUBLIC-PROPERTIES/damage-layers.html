CombatSystemProjectile.DamageLayers

Declaration

public int DamageLayers { get; set; }


Description

Indicates which layers can be damaged.

Upon spawning a projectile, it is crucial to set the layers that the projectile should damage in the event of a collision.
It's worth noting that the CharacterCombat class within the asset is already configured to handle this correctly in the event that a projectile is spawned by an attack.


The example below demonstrates the appropriate method for setting the property. 

using UnityEngine;

public class Example : MonoBehaviour
{
    // Note that the provided prefab should have the CombatSystemProjectile component attached.
    [SerializeField] private GameObject projectilePrefab;
    [SerializeField] private Transform spawnTransform;
    [SerializeField] private LayerMask damageableLayers;

    private void Update()
    {
        if (Input.GetKeyDown(KeyCode.X))
        {
            GameObject projectile = Instantiate(projectilePrefab, spawnTransform.position, Quaternion.identity);
            CombatSystemProjectile csProjectile = projectile.GetComponent<CombatSystemProjectile>();
            csProjectile.DamageLayers = damageableLayers;
            csProjectile.enabled = true;
        }
    }
}

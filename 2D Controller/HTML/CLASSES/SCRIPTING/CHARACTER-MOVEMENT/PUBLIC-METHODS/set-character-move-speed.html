CharacterMovement.SetCharacterMoveSpeed

Declaration

public float SetCharacterMoveSpeed(float value);


Description

Set a new movement speed (top speed) for your character.


This example, demonstrates how SetChacterMoveSpeed could be used: 

using UnityEngine;
using CharacterController2D.LocomotionSystem;

[RequireComponent(typeof(CharacterMovement))]
public class Example : MonoBehaviour 
{
    [SerializeField] private CharacterMovement characterMovement;
    [Tooltip("Assigned values greater than 1 will result in buffing the speed, while values below 1 will nerf the speed instead.")]
    [SerializeField] private float speedBuffMultiplier;
    
    public bool PlayerHasSpeedBuff { get; set; }

    private void Awake()
    {
        characterMovement = GetComponent<CharacterMovement>();
    }

    private void Update()
    {
        if (PlayerHasSpeedBuff)
        {
            // Change your character speed based on a multiplier set in the inspector.
            float newMovementSpeed = characterMovement.TopSpeed * speedBuffMultiplier;
            characterMovement.SetCharacterMoveSpeed(newMovementSpeed);
        }
        else
        {
            // Revert back to the default speed.
            characterMovement.SetCharacterMoveSpeed(characterMovement.TopSpeed);
        }
    }
}

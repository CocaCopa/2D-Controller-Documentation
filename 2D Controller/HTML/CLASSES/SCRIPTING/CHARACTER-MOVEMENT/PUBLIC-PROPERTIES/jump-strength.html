CharacterMovement.JumpStrength

Declaration

public float JumpStrength { get; set; }


Description

Modifies the strength of the jump.

In the example below, the character's jump will be increased if a power-up is collected. Add a new GameObject to your scene with a collider of your choice set as trigger, and set its tag to JumpPowerUp. Then, attach the provided script to your character. 

using UnityEngine;
using CharacterController2D.LocomotionSystem;

[RequireComponent(typeof(CharacterMovement))]
public class Example : MonoBehaviour 
{
    private CharacterMovement characterMovement;

    private void Awake()
    {
        characterMovement = GetComponent<CharacterMovement>();
    }

    private void OnTriggerEnter2D(Collider2D collision)
    {
        if (collision.CompareTag("JumpPowerUp"))
        {
            characterMovement.JumpStrength *= 2f;
        }
    }
}

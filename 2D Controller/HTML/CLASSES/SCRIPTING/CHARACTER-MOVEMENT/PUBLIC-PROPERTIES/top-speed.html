CharacterMovement.TopSpeed

Declaration

public float TopSpeed { get; }


Description

The character's set movement speed.

This property represents the character's maximum movement speed within the game world. It defines the upper limit of how fast the character can move when traversing the environment. Adjusting this property allows you to fine-tune the character's mobility and responsiveness, providing control over the overall pace and fluidity of movement during gameplay. 

using UnityEngine;

[RequireComponent(typeof(CharacterMovement))]
public class Example : MonoBehaviour 
{
    private CharacterMovement characterMovement;

    private void Awake()
    {
        characterMovement = GetComponent<CharacterMovement>();
        Debug.Log("The maximum speed that your character has been set to reach is: " + characterMovement.TopSpeed + " Unity units per second");
    }
}

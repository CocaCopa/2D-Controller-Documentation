CharacterEnvironmentalQuery.HeadCollisionCheck

Declaration

public bool HeadCollisionCheck();


Returns

True, if there is a collider found in front of the character's head, otherwise false.


Description

Checks if there is a collider in front of the character's head.

This function performs a Physics2D.BoxCast in the character's transform.right direction, to check for any colliders that are in front of the character's head.


This is an example of how you can use this function in your script: 

using UnityEngine;

public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.HeadCollisionCheck())
        {
            Debug.Log("There is a collider in front of your character's head.");
        }
        else
        {
            Debug.Log("No colliders found.");
        }
    }
}

Declaration

public bool HeadCollisionCheck(float distance);


Parameters
distance Override the default cast distance.

Returns

True, if there is a collider found in front of the character's head at a specified distance, otherwise false.


Description

With this overload you can override the default cast distance.


Here is an example of how you can use this overload in your script: 

using UnityEngine;

public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private float distance = 4f;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.HeadCollisionCheck(distance))
        {
            Debug.Log("There is a collider in front of your character's head at the maximum of " + distance + " Unity units away.");
        }
        else
        {
            Debug.Log("There are no colliders at the specified distance.");
        }
    }
}

Declaration

public bool HeadCollisionCheck(RaycastHit2D hitInfo, float distance);


Parameters
hitInfo If true is returned, hitInfo will contain more information about where the closest collider was hit.
distance Override the default cast distance.

Returns

True, if there is a collider found in front of the character's head at a specified distance, otherwise false.


Description

Same as above but it also returns information about the collider hit.


Here is an example of how you can use this overload in your script: 

using UnityEngine;

public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private float distance = 4f;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.HeadCollisionCheck(out RaycastHit2D hitInfo, distance))
        {
            Debug.Log("Collider: " + hitInfo.transform.name + " found in front of your character's head at the maximum of " + distance + " Unity units away.");
        }
        else
        {
            Debug.Log("There are no colliders at the specified distance.");
        }
    }
}

CharacterEnvironmentalQuery.GroundCheck

Declaration

public bool GroundCheck();


Returns

True, if the character is grounded, otherwise false.


Description

Checks if the character steps on a platform.

This function performs a Physics2D.BoxCast in the downward direction, taking into account the properties of the activeCollider.


This example demonstrates how you can use GroundCheck it in your code: 

using UnityEngine;

[RequireComponent(typeof(CharacterEnvironmentalQuery))]
public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.GroundCheck())
        {
            Debug.Log("Your character is grounded.");
        }
        else
        {
            Debug.Log("Your character is in the air.");
        }
    }
}

Declaration

public bool GroundCheck(float originOffset, float castDistance);


Parameters
originOffset Offsets the origin of the cast away from the character's right direction.
castDistance Overrides the default cast distance.

Returns

True, if the character is grounded, otherwise false.


Description

With this overload, you can override both the origin and the default distance of the cast.


Here is an example of how you can use this overload: 

using UnityEngine;

[RequireComponent(typeof(CharacterEnvironmentalQuery))]
public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private float offset = 2f;
    [SerializeField] private float distance = 4f;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.GroundCheck(offset, distance))
        {
            Debug.Log("Collider found " + offset + " Unity units away " + " at the maximum of " + distance + " Unity units below.");
        }
        else
        {
            Debug.Log("There are no colliders at the specified 'origin' and/or 'distance'.");
        }
    }
}

Declaration

public bool GroundCheck(float originOffset, float castDistance, RaycastHit2D hitInfo);


Parameters
originOffset Offsets the origin of the cast away from the character's right direction.
castDistance Overrides the default cast distance.
hitInfo If true is returned, hitInfo will contain more information about where the closest collider was hit.

Returns

True, if the character is grounded, otherwise false.


Description

Same as above, but it also returns information about the collider hit.


Here is an example of how you can use this overload: 

using UnityEngine;

[RequireComponent(typeof(CharacterEnvironmentalQuery))]
public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private float offset = 2f;
    [SerializeField] private float distance = 4f;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.GroundCheck(offset, distance, out RaycastHit2D hitInfo))
        {
            Debug.Log("Collider: " + hitInfo.transform.name + " found " + offset + " Unity units away " + " at the maximum of " + distance + " Unity units below.");
        }
        else
        {
            Debug.Log("There are no colliders at the specified 'origin' and/or 'distance'.");
        }
    }
}

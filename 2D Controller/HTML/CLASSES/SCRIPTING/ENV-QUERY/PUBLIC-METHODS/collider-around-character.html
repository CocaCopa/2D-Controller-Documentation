CharacterEnvironmentalQuery.ColliderAroundCharacter

Declaration

public bool ColliderAroundCharacter(bool overlapBox, float range);


Parameters
overlapBox False, will cast an OverlapCapsule instead.
range This value represents a percentage of the character's collider, to be used as the size of the cast.

Returns

True, if the overlap cast finds any colliders, otherwise false.


Description

Searches for any collider in the specified range around the character based on the character's collider.

You can use this function to search for any colliders around your character in the game environment in the specified distance.


Method use example goes here. 

using UnityEngine;

[RequireComponent(typeof(CharacterEnvironmentalQuery))]
public class Example : Monobehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private float castRange = 0.2f;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent⟨CharacterEnvironmentalQuery⟩();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.ColliderAroundCharacter(overlapBox: false, castRange))
        {
            Debug.Log("Collider found in " + castRange + " Unity units away");
        }
    }
}

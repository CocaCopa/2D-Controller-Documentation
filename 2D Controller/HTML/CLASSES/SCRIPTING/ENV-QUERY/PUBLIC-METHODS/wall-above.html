CharacterEnvironmentalQuery.WallAbove

Declaration

public bool WallAbove();


Returns

True, if a collider is found above the character's head, otherwise false.


Description

Checks if there is a wall above of the character's head.

This function performs a Physics2D.BoxCast in the upward direction, taking into account the size and position of the activeCollider.


Method use example goes here. 

using UnityEngine;

[RequireComponent(typeof(CharacterEnvironmentalQuery))]
public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.WallAbove())
        {
            Debug.Log("There is a wall above the charater's head");
        }
        else
        {
            Debug.Log("There is not collider at the specified distance");
        }
    }
}

Declaration

public bool WallAbove(float distance);


Returns

True, if a collider is found at the specified distance in the upward direction, otherwise false.


Parameters
distance Overrides the default cast distance.

Description

Checks if there is a wall above of the character's head.

Same as the above, but with this overload you can override the default cast distance.


This is an example of how you can use this method in your script. 

using UnityEngine;

[RequireComponent(typeof(CharacterEnvironmentalQuery))]
public class Example : MonoBehaviour
{
    [SerializeField] private CapsuleCollider2D m_Collider;
    [SerializeField] private float distance = 5f;

    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (envQuery.WallAbove(distance))
        {
            Debug.Log("There is a wall above the charater's head at the maximum of " + distance + " Unity units above");
        }
        else
        {
            Debug.Log("There is not a collider at the specified distance");
        }
    }
}

EntityHealth.IsRegeneratingHealth

Declaration

public bool IsRegeneratingHealth { get; }


Description

Indicates whether the entity is regenerating health points.

This property indicates whether the entity is currently in the process of regenerating health points. When set to true, it signifies that the entity's health points are actively being restored over time through a regeneration effect. You can utilize this property to monitor the health regeneration state of the entity, enabling the implementation of gameplay mechanics or visual feedback tailored to the regenerative process. 

using UnityEngine;

public class Example : MonoBehaviour
{
    private EntityHealth healthScript;

    private void Awake()
    {
        healthScript = GetComponent<EntityHealth>();
    }

    private void Update()
    {
        if (healthScript.IsRegeneratingHealth)
        {
            Debug.Log("The entity is regenerating health.");
        }
    }
}

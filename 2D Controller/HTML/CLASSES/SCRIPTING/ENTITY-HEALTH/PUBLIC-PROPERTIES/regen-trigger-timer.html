EntityHealth.RegenTriggerTimer

Declaration

public float RegenTriggerTimer { get; }


Description

Triggers health regeneration when this value reaches 0.

This property is pivotal in controlling the timing of health recovery mechanics within the gameplay. Once the timer reaches 0, it triggers the activation of health regeneration, allowing the entity to gradually restore its health points over time.


You can utilize this property to generate visual cues for the player, indicating when the regeneration is about to take effect or other related behaviours. 

using UnityEngine;
using CharacterController2D.CombatSystem;

[RequireComponent(typeof(EntityHealth))]
public class Example : MonoBehaviour
{
    private EntityHealth healthScript;

    private void Awake()
    {
        healthScript = GetComponent<EntityHealth>();
    }

    private void Update()
    {
        if (healthScript.CanRegenHealth && healthScript.CurrentHealthPoints != healthScript.MaxHealthPoints)
        {
            Debug.Log("Health regeneration will take effect in: " + healthScript.RegenTriggerTimer + " seconds");
        }
    }
}

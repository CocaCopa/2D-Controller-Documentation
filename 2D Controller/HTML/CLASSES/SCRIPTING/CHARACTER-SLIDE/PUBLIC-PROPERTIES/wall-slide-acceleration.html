CharacterSlide.WallSlideAcceleration

Declaration

public float WallSlideAcceleration { get; set; }


Description

The speed at which the character accelerates during a wall slide.

You can adjust this value based on the surface onto which the character is sliding to vary surface friction levels.


In this example, the value of the property will change depending on the surface the character is sliding on.
Note that the wall surfaces must be tagged as "LowFriction" and "NormalFriction" in order for the example below to work correctly. 

using UnityEngine;

[RequireComponent(typeof(CharacterSlide))]
public class Example : MonoBehaviour 
{
    [SerializeField] private float minimumSlideSpeed;
    
    private CharacterSlide characterSlide;
    
    private void Awake()
    {
        characterSlide = GetComponent<CharacterSlide>();
    }

    private void Update()
    {
        // vv Your code for character horizontal movement vv
        // Code...
        
        if (Input.GetKey(Keycode.DownArrow))
        {
            characterSlide.FloorSlide(minimumSlideSpeed);
        }
    }
    
    private void OnCollisionEnter2D(Collision2D col)
    {
        if (col.gameObject.CompareTag("LowFriction"))
        {
            characterSlide.WallSlideAcceleration = characterSlide.DefaultWallAcceleration / 2f;
        }

        if (col.gameObject.CompareTag("NormalFriction"))
        {
            characterSlide.WallSlideAcceleration = characterSlide.DefaultWallAcceleration;
        }
    }
}

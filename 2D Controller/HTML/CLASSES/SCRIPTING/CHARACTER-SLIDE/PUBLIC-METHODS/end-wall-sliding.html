CharacterSlide.EndWallSliding

Declaration

public float EndWallSliding();


Description

Resets certain values when the character exits the wall slide.

This function should be called after calling StartWallSliding in order for the character state to be reset to their default state.


This is an example of how you can call the function in your script, utilizing the CharacterEnvironmentalQuery class, for convinience: 

using UnityEngine;
using CharacterController2D.LocomotionSystem;
using CharacterController2D.Core;

[RequireComponent(typeof(CharacterSlide), typeof(CharacterEnvironmentalQuery), typeof(CapsuleCollider2D))]
public class Example : MonoBehaviour
{
    private CapsuleCollider2D m_Collider;
    private CharacterSlide characterSlide;
    private CharacterEnvironmentalQuery envQuery;

    private void Awake()
    {
        m_Collider = GetComponent<CapsuleCollider2D>();
        characterSlide = GetComponent<CharacterSlide>();
        envQuery = GetComponent<CharacterEnvironmentalQuery>();
        envQuery.SetActiveCollider(m_Collider);
    }

    private void Update()
    {
        if (!envQuery.GroundCheck() && envQuery.WallInFront())
        {
            characterSlide.StartWallSliding();
        }
        else
        {
            characterSlide.EndWallSliding();
        }
    }
}
